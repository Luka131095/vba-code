Sub CalculateCostSummary()
    Dim ws As Worksheet
    Set ws = ThisWorkbook.Sheets("Sheet1")
    Dim lastRow As Long
    lastRow = ws.Cells(ws.Rows.Count, "A").End(xlUp).Row
    ws.Range("B" & lastRow + 1).Value = Application.WorksheetFunction.Sum(ws.Range("B2:B" & lastRow))
    MsgBox "Cost summary calculated!", vbInformation
End Sub
